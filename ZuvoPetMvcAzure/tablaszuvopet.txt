DROP TABLE IF EXISTS HISTORIASEXITO;
DROP TABLE IF EXISTS NOTIFICACIONES;
DROP TABLE IF EXISTS MENSAJES;
DROP TABLE IF EXISTS MASCOTAS;

DROP TABLE IF EXISTS ADOPTANTES;
DROP TABLE IF EXISTS PERFIL_USUARIO;
DROP TABLE IF EXISTS USUARIOS;

-- 📌 TABLA Usuarios (Base de autenticación)
CREATE TABLE USUARIOS (
    Id INT PRIMARY KEY,  -- Identificador único del usuario
    NombreUsuario NVARCHAR(50) UNIQUE NOT NULL,  -- Nombre de usuario (único)
    Email NVARCHAR(255) UNIQUE NOT NULL,  -- Email (único)
    ContrasenaLimpia NVARCHAR(255) NOT NULL,  -- Contraseña en texto claro (se utiliza temporalmente para encriptación)
    ContrasenaEncriptada VARBINARY(MAX) NOT NULL,  -- Contraseña encriptada (almacenada como byte array)
    Salt NVARCHAR(255) NOT NULL,  -- Salt para la contraseña
    TipoUsuario NVARCHAR(20) NOT NULL CHECK (TipoUsuario IN ('Adoptante', 'Refugio')),  -- Tipo de usuario
    FechaRegistro DATETIME DEFAULT GETDATE(),  -- Fecha de creación del usuario
);
GO

-- 📌 TABLA Perfil_Usuario (Visible para el exterior)
CREATE TABLE PERFIL_USUARIO (
    Id INT PRIMARY KEY,  -- Identificador único del perfil de usuario
    IdUsuario INT UNIQUE NOT NULL,  -- Relación con la tabla USUARIOS
    FotoPerfil NVARCHAR(255) NULL,  -- Ruta de la imagen de perfil
    Descripcion NVARCHAR(500) NULL,  -- Descripción del usuario (opcional)
    FOREIGN KEY (IdUsuario) REFERENCES USUARIOS(Id)  -- Relación con la tabla USUARIOS
);
GO

-- 📌 TABLA Adoptantes
CREATE TABLE ADOPTANTES (
    Id INT PRIMARY KEY,  -- Identificador único del adoptante
    IdUsuario INT UNIQUE NOT NULL,  -- Relación con la tabla USUARIOS
    Nombre NVARCHAR(255) NOT NULL,  -- Nombre real del adoptante
    TipoVivienda NVARCHAR(100) NULL,  -- Tipo de vivienda
    TieneJardin BIT NOT NULL DEFAULT 0,  -- Tiene jardín o no
    OtrosAnimales BIT NOT NULL DEFAULT 0,  -- Tiene otras mascotas o no
    RecursosDisponibles NVARCHAR(255) NULL,  -- Recursos disponibles para la mascota
    TiempoEnCasa NVARCHAR(255) NULL,  -- Tiempo que pasa en casa
    FOREIGN KEY (IdUsuario) REFERENCES USUARIOS(Id)  -- Relación con USUARIOS
);
GO

-- 📌 TABLA Refugios
CREATE TABLE REFUGIOS (
    Id INT PRIMARY KEY,  -- Identificador único del refugio
    IdUsuario INT UNIQUE NOT NULL,  -- Relación con la tabla USUARIOS
    NombreRefugio NVARCHAR(255) NOT NULL,  -- Nombre del refugio
    ContactoRefugio NVARCHAR(255) NOT NULL,  -- Contacto del refugio (Teléfono/Email)
    CantidadAnimales INT NOT NULL,  -- Número de animales actualmente en el refugio
    CapacidadMaxima INT NOT NULL,  -- Capacidad máxima de animales en el refugio
    Latitud FLOAT NULL,  -- Latitud del refugio
    Longitud FLOAT NULL,  -- Longitud del refugio
    FOREIGN KEY (IdUsuario) REFERENCES USUARIOS(Id)  -- Relación con USUARIOS
);
GO

-- 📌 TABLA Mascotas
CREATE TABLE MASCOTAS (
    Id INT PRIMARY KEY,  -- Identificador único de la mascota
    Nombre NVARCHAR(100) NOT NULL,  -- Nombre de la mascota
    Especie NVARCHAR(50) NOT NULL,  -- Especie de la mascota (Perro, Gato, etc.)
    FechaNacimiento DATE NOT NULL,  -- Se usa para calcular edad de forma precisa
    Tamano NVARCHAR(50) NOT NULL,  -- Tamano de la mascota (Pequeño, Mediano, Grande)
    Sexo NVARCHAR(20) CHECK (Sexo IN ('Macho', 'Hembra')),  -- Sexo de la mascota
    Castrado BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota está esterilizada (1 = Sí, 0 = No)
    Vacunado BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota tiene vacunas al día
    Desparasitado BIT DEFAULT 0 NOT NULL,  -- Indica si ha sido desparasitada
    Sano BIT DEFAULT 1 NOT NULL,  -- Indica si la mascota está sana
    Microchip BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota tiene microchip
    CompatibleConNinos BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota es compatible con ninos
    CompatibleConAdultos BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota es compatible con adultos
    CompatibleConOtrosAnimales BIT DEFAULT 0 NOT NULL,  -- Indica si la mascota es compatible con otros animales
    Personalidad NVARCHAR(MAX) NULL, -- Lista de adjetivos separados por comas ("Cariñoso, Sociable")
    Descripcion TEXT NULL,  -- Descripción general de la mascota
    Estado NVARCHAR(50) DEFAULT 'Disponible' CHECK (Estado IN ('Disponible', 'Adoptado')),  -- Estado de la mascota (Disponible, Adoptado)
    Latitud FLOAT NULL,  
    Longitud FLOAT NULL, 
    IdRefugio INT NULL,  -- ID del usuario (refugio o persona) que la registró
    Foto NVARCHAR(255) NULL,  -- Ruta de la imagen de la mascota
    Vistas INT DEFAULT 0 NOT NULL,  -- Contador de visualizaciones del perfil de la mascota
    FechaRegistro DATETIME DEFAULT GETDATE(),  -- Fecha en la que la mascota fue registrada
    FOREIGN KEY (IdRefugio) REFERENCES USUARIOS(Id)  -- Relación con el usuario (adoptante o refugio) que la registró
);
GO

-- 📌 TABLA Favoritos
CREATE TABLE FAVORITOS (
    Id INT PRIMARY KEY,  -- Identificador único del favorito
    IdAdoptante INT NOT NULL,  -- Usuario que marcó la mascota como favorita
    IdMascota INT NOT NULL,  -- ID de la mascota que fue marcada como favorita
    FechaAgregado DATETIME DEFAULT GETDATE(),  -- Fecha en que se agregó a favoritos
    FOREIGN KEY (IdAdoptante) REFERENCES ADOPTANTES(Id),  -- Relación con la tabla de usuarios
    FOREIGN KEY (IdMascota) REFERENCES MASCOTAS(Id),  -- Relación con la tabla de mascotas
    UNIQUE (IdAdoptante, IdMascota)  -- Evita que un usuario marque la misma mascota más de una vez
);
GO

-- 📌 TABLA Solicitudes de Adopción
CREATE TABLE SOLICITUDESADOPCION (
    Id INT PRIMARY KEY,  -- Identificador único de la solicitud de adopción
    IdAdoptante INT NOT NULL,  -- ID del usuario que solicita la adopción
    IdMascota INT NOT NULL,  -- ID de la mascota que se desea adoptar
    Estado NVARCHAR(50) DEFAULT 'Pendiente' CHECK (Estado IN ('Pendiente', 'Aprobada', 'Rechazada')),  -- Estado de la solicitud ('Pendiente', 'Aprobada', 'Rechazada')
    ExperienciaPrevia BIT,  -- Indica si el usuario ha tenido mascotas antes (1 = Sí, 0 = No)
    TipoVivienda NVARCHAR(100),  -- Tipo de vivienda del adoptante (Casa, Departamento, etc.)
    OtrosAnimales BIT,  -- Indica si el adoptante tiene otras mascotas (1 = Sí, 0 = No)
    Recursos NVARCHAR(255),  -- Recursos disponibles para el cuidado de la mascota (Ej: "Jardín, Patio, Tiempo Completo")
    TiempoEnCasa NVARCHAR(255),  -- Tiempo que el adoptante pasa en casa (Ej: "Todo el día, Solo noches")
    FechaSolicitud DATETIME DEFAULT GETDATE(),  -- Fecha en que se realizó la solicitud
    FechaRespuesta DATETIME,  -- Fecha en que se aprobó/rechazó la solicitud (NULL si aún está pendiente)
    FOREIGN KEY (IdAdoptante) REFERENCES ADOPTANTES(Id),  -- Relación con el usuario adoptante
    FOREIGN KEY (IdMascota) REFERENCES MASCOTAS(Id)  -- Relación con la mascota que se desea adoptar
);
GO 

-- 📌 TABLA Historias de Éxito
CREATE TABLE HISTORIASEXITO (
    Id INT PRIMARY KEY,  -- Identificador único de la historia de éxito
    IdAdoptante INT NOT NULL,  -- ID del usuario que publica la historia
    IdMascota INT NOT NULL,  -- ID de la mascota protagonista de la historia
    Titulo NVARCHAR(255) NOT NULL,  -- Título de la historia de éxito
    Descripcion TEXT NOT NULL,  -- Relato completo de la historia de adopción
    Foto NVARCHAR(255) NULL,  -- Ruta o URL de una imagen representativa
    FechaPublicacion DATETIME DEFAULT GETDATE(),  -- Fecha en que se publicó la historia
    Estado NVARCHAR(50) DEFAULT 'Pendiente' CHECK (Estado IN ('Pendiente', 'Aprobada', 'Rechazada')),  -- Estado de la historia ('Pendiente', 'Aprobada', 'Rechazada')
    FOREIGN KEY (IdAdoptante) REFERENCES ADOPTANTES(Id),  -- Relación con el usuario que adopta
    FOREIGN KEY (IdMascota) REFERENCES MASCOTAS(Id)  -- Relación con la mascota adoptada
);
GO

-- 📌 TABLA Likes en Historias
CREATE TABLE LIKESHISTORIAS (
    Id INT PRIMARY KEY,  -- Identificador único del "like"
    IdHistoria INT NOT NULL,  -- ID de la historia que recibe la reacción
    IdUsuario INT NOT NULL,  -- ID del usuario que da la reacción
    TipoReaccion NVARCHAR(20) NOT NULL CHECK (TipoReaccion IN ('MeGusta', 'MeEncanta', 'Inspirador', 'Solidario', 'Asombroso')),  -- Tipo de reacción (MeGusta, MeEncanta, etc.)
    Fecha DATETIME DEFAULT GETDATE(),  -- Fecha y hora en que se realizó la reacción
    FOREIGN KEY (IdHistoria) REFERENCES HISTORIASEXITO(Id),  -- Relación con la historia
    FOREIGN KEY (IdUsuario) REFERENCES USUARIOS(Id),  -- Relación con el usuario
    UNIQUE (IdHistoria, IdUsuario, TipoReaccion)  -- Evita que un usuario dé el mismo tipo de "like" más de una vez a una historia
);
GO

-- 📌 TABLA Notificaciones (Para SignalR)
CREATE TABLE NOTIFICACIONES (
    Id INT PRIMARY KEY,  -- Identificador único de la notificación
    IdUsuario INT NOT NULL,  -- ID del usuario que recibe la notificación
    Mensaje NVARCHAR(255) NOT NULL,  -- Contenido del mensaje de la notificación
	Tipo NVARCHAR(50) NOT NULL CHECK (Tipo IN ('Solicitud', 'Aprobación', 'Mensaje', 'Sistema')),  -- Tipo de notificación
    Url NVARCHAR(255) NULL,  -- URL opcional para redirigir al usuario
    Fecha DATETIME DEFAULT GETDATE(),  -- Fecha y hora en que se generó la notificación
    Leido BIT DEFAULT 0,  -- Estado de la notificación (0 = No leída, 1 = Leída)
    FOREIGN KEY (IdUsuario) REFERENCES USUARIOS(Id)  -- Relación con el usuario receptor de la notificación
);
GO

-- 📌 TABLA Mensajes (Para SignalR)
CREATE TABLE MENSAJES (
    Id INT PRIMARY KEY,  -- Identificador único del mensaje
    IdEmisor INT NOT NULL,  -- ID del usuario que envía el mensaje
    IdReceptor INT NOT NULL,  -- ID del usuario que recibe el mensaje
    Contenido TEXT NOT NULL,  -- Contenido del mensaje enviado
    Fecha DATETIME DEFAULT GETDATE(),  -- Fecha y hora en que se envió el mensaje
	Leido BIT DEFAULT 0, -- Estado del mensaje (0 = No leída, 1 = Leída)
    FOREIGN KEY (IdEmisor) REFERENCES USUARIOS(Id),  -- Relación con el usuario que envió el mensaje
    FOREIGN KEY (IdReceptor) REFERENCES USUARIOS(Id)  -- Relación con el usuario que recibió el mensaje
);
GO

CREATE OR ALTER VIEW V_AdoptantePerfil AS
SELECT 
    U.Id AS IdUsuario,
    U.NombreUsuario,
    U.Email,
    P.FotoPerfil,
    P.Descripcion,
    A.Nombre,
    A.TipoVivienda,
    A.TieneJardin,
    A.OtrosAnimales,
    A.RecursosDisponibles,
    A.TiempoEnCasa
FROM USUARIOS U
JOIN PERFIL_USUARIO P ON U.Id = P.IdUsuario
JOIN ADOPTANTES A ON U.Id = A.IdUsuario
WHERE U.TipoUsuario = 'Adoptante';
GO

CREATE OR ALTER PROCEDURE SP_PERFILADOPTANTE
(@idusuario int)
as
	select * from V_AdoptantePerfil
	where IdUsuario = @idusuario;
go

CREATE OR ALTER VIEW V_RefugioPerfil AS
SELECT 
    U.Id AS IdUsuario,
    U.NombreUsuario,
    U.Email,
    P.FotoPerfil,
    P.Descripcion,
    R.NombreRefugio,
    R.ContactoRefugio,
    R.CantidadAnimales,
    R.CapacidadMaxima,
    R.Latitud,
	R.Longitud
FROM USUARIOS U
JOIN PERFIL_USUARIO P ON U.Id = P.IdUsuario
JOIN REFUGIOS R ON U.Id = R.IdUsuario
WHERE U.TipoUsuario = 'Refugio';
GO
CREATE OR ALTER PROCEDURE SP_PERFILREFUGIO
(@idusuario int)
as
	select * from V_RefugioPerfil
	where IdUsuario = @idusuario;
go

CREATE OR ALTER VIEW V_MascotasFavoritas
AS
SELECT 
	m.Id,
	m.Nombre,
	m.Especie,
	DATEDIFF(MONTH, m.FechaNacimiento, GETDATE()) AS Edad,
	m.Tamano,
	m.Sexo,
	m.Foto,
	f.IdAdoptante,
	f.FechaAgregado
FROM
 MASCOTAS m
 INNER JOIN
 FAVORITOS f on m.Id = f.IdMascota
 GO

create or alter procedure SP_OBTENERMASCOTASFAVORITAS
 (@idadoptante int)
 as
	select * from V_MascotasFavoritas
	where IdAdoptante = @idadoptante
	order by FechaAgregado desc
 go

CREATE OR ALTER VIEW V_MascotasAdoptadas
AS
SELECT 
    m.Id,
    m.Nombre,
    m.Especie,
    DATEDIFF(MONTH, m.FechaNacimiento, GETDATE()) AS Edad,
    m.Tamano,
    m.Sexo,
    m.Foto,
    s.IdAdoptante,
    s.FechaRespuesta AS FechaAdopcion
FROM
    MASCOTAS m
    INNER JOIN SOLICITUDESADOPCION s ON m.Id = s.IdMascota
WHERE
    s.Estado = 'Aprobada'
GO

CREATE OR ALTER PROCEDURE SP_OBTENERMASCOTASADOPTADAS
    (@idadoptante int)
AS
    SELECT * FROM V_MascotasAdoptadas
    WHERE IdAdoptante = @idadoptante
    ORDER BY FechaAdopcion DESC
GO
